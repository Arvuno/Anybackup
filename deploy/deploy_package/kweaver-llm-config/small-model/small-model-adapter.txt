import aiohttp
import json
import uuid
import os
from typing import List, Dict, Any

# 假设 API Key 存储在环境变量中，或者直接替换为您的 Key
DASHSCOPE_API_KEY = "sk-a6cf2c1fe92d47a398dc53781e84bb5e"
# DASHSCOPE_API_KEY = os.getenv("DASHSCOPE_API_KEY", "YOUR_DASHSCOPE_API_KEY_HERE")

async def main(query: str, documents: List[str]) -> Dict[str, Any]:
    """
    异步调用阿里云 DashScope Rerank API，对文档进行排序。
    Args:
        query (str): 搜索查询字符串。
        documents (List[str]): 待排序的文档列表。
    Returns:
        Dict[str, Any]: 包含重排序结果和用量信息的响应体。
    """
    # 阿里云 DashScope Rerank API 的 URL
    url = "https://dashscope.aliyuncs.com/api/v1/services/rerank/text-rerank/text-rerank"
 

    # HTTP 请求头
    headers = {
        "Authorization": f"Bearer {DASHSCOPE_API_KEY}",
        "Content-Type": "application/json",
    }
  

    # 请求体 payload，适配阿里云的 JSON 格式
    payload = {
        "model": "qwen3-vl-rerank",  # 使用 curl 示例中指定的模型
        "input": {
            "query": query,
            "documents": documents
        },
        "parameters": {
            # 这里的参数根据您 curl 示例中固定值设置
            "return_documents": True,
            "top_n": 2,
            "instruct": "Given a web search query, retrieve relevant passages that answer the query."
        }
    }
    print(f"INFO: Sending Rerank request to DashScope for query: {query}")


    # 使用 aiohttp 异步发送 POST 请求
    async with aiohttp.ClientSession() as session:
        async with session.post(url, json=payload, headers=headers) as resp:
            # 检查响应状态码
            if resp.status != 200:
                error_detail = await resp.text()
                raise Exception(f"DashScope API failed with status {resp.status}. Details: {error_detail}")
           
            # 解析 JSON 响应
            dashscope_scores = await resp.json()
          
    # 提取核心信息，构建一个标准化响应体 (模仿您提供的示例格式)
    # 阿里云 DashScope 响应通常包含 'output' 和 'usage' 字段
    results = dashscope_scores.get("output", {}).get("results", [])
    usage = dashscope_scores.get("usage", {})
   
    response = {
        "id": str(uuid.uuid4()),
        "model": payload["model"],
        "object": "rerank_result",
        "results": results, # 包含重排序后的文档和分数列表
        "usage": usage      # 包含 token 统计信息
    }
    return response
